1. Bulk File Organizer

Saves: 30 minutes/week of manual file sorting

Automatically sorts files into folders by extension. Downloads folder → organized in seconds.

import os, shutil
from pathlib import Path

DOWNLOADS = Path.home() / "Downloads"
CATEGORIES = {
    "Images": [".jpg", ".jpeg", ".png", ".gif", ".webp", ".svg"],
    "Documents": [".pdf", ".docx", ".xlsx", ".pptx", ".txt", ".csv"],
    "Audio": [".mp3", ".wav", ".flac", ".aac"],
    "Video": [".mp4", ".mov", ".avi", ".mkv"],
    "Archives": [".zip", ".rar", ".7z", ".tar.gz"],
    "Code": [".py", ".js", ".html", ".css", ".json"],
}

for file in DOWNLOADS.iterdir():
    if file.is_file():
        ext = file.suffix.lower()
        for category, extensions in CATEGORIES.items():
            if ext in extensions:
                dest = DOWNLOADS / category
                dest.mkdir(exist_ok=True)
                shutil.move(str(file), str(dest / file.name))
                print(f"Moved: {file.name} → {category}/")
                break

2. CSV Merger — Combine 100 Files in 2 Seconds

Saves: 20 minutes per merge operation

Got 50 CSV exports from different sources? Merge them all into one clean file.

import pandas as pd
from pathlib import Path
import glob

csv_files = glob.glob("data/*.csv")
dfs = []

for file in csv_files:
    df = pd.read_csv(file)
    print(f"Loaded: {file} ({len(df)} rows)")
    dfs.append(df)

merged = pd.concat(dfs, ignore_index=True)
merged.to_csv("merged_output.csv", index=False)
print(f"Done! {len(merged)} total rows → merged_output.csv")

3. Automated Email Reports

Saves: 15 minutes per report

Send automated daily/weekly reports to your team or clients.

import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart
from datetime import date

SMTP_SERVER = "smtp.gmail.com"
SMTP_PORT = 587
SENDER = "your_email@gmail.com"
PASSWORD = "your_app_password"  # Use App Password, not regular password

def send_report(to_email, subject, body, attachment_path=None):
    msg = MIMEMultipart()
    msg["From"] = SENDER
    msg["To"] = to_email
    msg["Subject"] = subject
    msg.attach(MIMEText(body, "plain"))

    if attachment_path:
        with open(attachment_path, "rb") as f:
            attach = MIMEText(f.read(), "base64", "utf-8")
            attach.add_header("Content-Disposition", f"attachment; filename={attachment_path}")
            msg.attach(attach)

    with smtplib.SMTP(SMTP_SERVER, SMTP_PORT) as server:
        server.starttls()
        server.login(SENDER, PASSWORD)
        server.send_message(msg)

send_report(
    "boss@company.com",
    f"Daily Report — {date.today()}",
    "Hi, here's today's summary:\n\n- Sales: $1,245\n- New leads: 12\n- Support tickets: 4",
    "daily_report.csv"
)

4. PDF Merger & Splitter

Saves: 10 minutes per PDF operation

Combine 20 invoices into one PDF. Or split one large PDF into individual pages.

from PyPDF2 import PdfMerger, PdfReader, PdfWriter

# Merge PDFs
merger = PdfMerger()
for pdf in ["report1.pdf", "report2.pdf", "report3.pdf"]:
    merger.append(pdf)
merger.write("combined_report.pdf")
merger.close()
print("Merged into combined_report.pdf")

# Split PDF into individual pages
reader = PdfReader("large_document.pdf")
for i, page in enumerate(reader.pages):
    writer = PdfWriter()
    writer.add_page(page)
    writer.write(f"page_{i+1}.pdf")
print(f"Split into {len(reader.pages)} files")

6. Duplicate File Finder

Saves: Hours of manual cleanup + GBs of disk space

import os, hashlib
from pathlib import Path
from collections import defaultdict

def file_hash(filepath):
    hasher = hashlib.md5()
    with open(filepath, "rb") as f:
        for chunk in iter(lambda: f.read(4096), b""):
            hasher.update(chunk)
    return hasher.hexdigest()

def find_duplicates(directory):
    hashes = defaultdict(list)
    for filepath in Path(directory).rglob("*"):
        if filepath.is_file():
            hashes[file_hash(filepath)].append(filepath)

    for h, files in hashes.items():
        if len(files) > 1:
            print(f"\nDuplicate group ({len(files)} files):")
            for f in files:
                size_mb = os.path.getsize(f) / (1024*1024)
                print(f"  {f} ({size_mb:.1f} MB)")

find_duplicates("C:/Users/Admin/Documents")

7. Automated Excel Report Generator

Saves: 30+ minutes per report

Pull data from multiple sources, run calculations, create charts, and save a formatted Excel report.

import pandas as pd
from openpyxl import Workbook
from openpyxl.chart import BarChart, Reference
from openpyxl.styles import Font, PatternFill

# Read data
sales = pd.read_csv("sales_data.csv")
sales["revenue"] = sales["quantity"] * sales["price"]

# Summarize
summary = sales.groupby("product")["revenue"].sum().sort_values(ascending=False)

# Write to Excel with formatting
wb = Workbook()
ws = wb.active
ws.title = "Sales Report"

ws["A1"] = "Sales Report"
ws["A1"].font = Font(size=16, bold=True)

ws["A3"] = "Product"; ws["B3"] = "Revenue"
for cell in [ws["A3"], ws["B3"]]:
    cell.font = Font(bold=True, color="ffffff")
    cell.fill = PatternFill(start_color="1e3a5f", fill_type="solid")

for i, (product, rev) in enumerate(summary.items()):
    ws[f"A{4+i}"] = product
    ws[f"B{4+i}"] = rev
    ws[f"B{4+i}"].number_format = '$#,##0'

# Add chart
chart = BarChart()
data = Reference(ws, min_col=2, min_row=3, max_row=3+len(summary))
chart.add_data(data, titles_from_data=True)
chart.set_categories(Reference(ws, min_col=1, min_row=4, max_row=3+len(summary)))
ws.add_chart(chart, "D3")

wb.save("sales_report.xlsx")
print("Report saved: sales_report.xlsx")

8. Image Bulk Resizer

Saves: 1+ hour per batch

Resize 500 product photos to web-optimized sizes in 30 seconds.

from PIL import Image
from pathlib import Path

INPUT_DIR = Path("original_photos")
OUTPUT_DIR = Path("resized_photos")
OUTPUT_DIR.mkdir(exist_ok=True)
MAX_WIDTH = 1200

for img_path in INPUT_DIR.glob("*"):
    if img_path.suffix.lower() in (".jpg", ".jpeg", ".png", ".webp"):
        img = Image.open(img_path)
        if img.width > MAX_WIDTH:
            ratio = MAX_WIDTH / img.width
            new_size = (MAX_WIDTH, int(img.height * ratio))
            img = img.resize(new_size, Image.LANCZOS)

        output_path = OUTPUT_DIR / f"{img_path.stem}.jpg"
        img.convert("RGB").save(output_path, "JPEG", quality=85, optimize=True)
        print(f"Resized: {img_path.name} → {output_path.name}")

9. Text File Batch Processor

Saves: 30+ minutes per batch edit

Find-and-replace across 100 text files, rename batches, extract specific lines.

from pathlib import Path

def batch_replace(directory, find_text, replace_text, file_pattern="*.txt"):
    """Replace text across all matching files in a directory."""
    for filepath in Path(directory).glob(file_pattern):
        content = filepath.read_text(encoding="utf-8")
        count = content.count(find_text)
        if count > 0:
            content = content.replace(find_text, replace_text)
            filepath.write_text(content, encoding="utf-8")
            print(f"Updated: {filepath.name} ({count} replacements)")

# Usage
batch_replace("marketing/copy/", "2025", "2026")
